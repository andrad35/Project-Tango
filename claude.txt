 //@version=5
  strategy("HL-HH ES Futures 15m", overlay=true, pyramiding=0,                   
  default_qty_type=strategy.fixed, default_qty_value=1, currency=currency.USD,
  commission_type=strategy.commission.cash_per_contract, commission_value=4.10,  
  slippage=1)     

  // ============================================================
  // INPUTS — PIVOT SETTINGS
  // ============================================================
  i_leftBars         = input.int(2,    "Pivot Left Bars",  minval=1, group="Pivot
   Settings")
  i_rightBars        = input.int(2,    "Pivot Right Bars", minval=1, group="Pivot
   Settings")

  // ============================================================
  // INPUTS — ENTRY SETTINGS
  // ============================================================
  i_entryMode        = input.string("Both", "Entry Mode", options=["Breakout
  Only", "Retest Only", "Both"], group="Entry Settings")
  i_retestTolerance  = input.float(0.5, "Retest Zone (ATR mult)", minval=0.1,
  maxval=2.0, step=0.1, group="Entry Settings")
  i_retestTimeout    = input.int(20,   "Retest Timeout (bars)", minval=5,
  maxval=100, group="Entry Settings")

  // ============================================================
  // INPUTS — RISK MANAGEMENT
  // ============================================================
  i_stopBuffer       = input.float(0.5, "Stop Buffer (ATR mult below HL)",
  minval=0.0, maxval=3.0, step=0.25, group="Risk Management")
  i_rrRatio          = input.float(2.0, "R:R Ratio (Take Profit)", minval=0.5,
  maxval=10.0, step=0.5, group="Risk Management")
  i_useTrailStop     = input.bool(false, "Use Trailing Stop", group="Risk
  Management")
  i_trailAtr         = input.float(1.5,  "Trail Distance (ATR mult)", minval=0.5,
   maxval=5.0, step=0.25, group="Risk Management")

  // ============================================================
  // INPUTS — FILTERS
  // ============================================================
  i_useEmaFilter     = input.bool(false, "EMA Trend Filter", group="Filters")
  i_emaLength        = input.int(50,     "EMA Length", minval=1, group="Filters")
  i_useVolFilter     = input.bool(false, "Volume Confirmation", group="Filters")
  i_volAvgLength     = input.int(20,     "Volume Average Length", minval=1,
  group="Filters")
  i_useAtrFilter     = input.bool(false, "Volatility Filter (ATR)",
  group="Filters")
  i_atrLength        = input.int(14,     "ATR Length", minval=1, group="Filters")
  i_atrAvgLength     = input.int(50,     "ATR Average Length", minval=1,
  group="Filters")
  i_atrMinPct        = input.float(0.8,  "Min ATR vs Average", minval=0.1,
  maxval=2.0, step=0.05, group="Filters")
  i_useSessionFilter = input.bool(false, "RTH Session Filter (ES only)",
  group="Filters")

  // ============================================================
  // INPUTS — DISPLAY
  // ============================================================
  i_showDots         = input.bool(true, "Show Pivot Dots", group="Display")
  i_showEma          = input.bool(true, "Show EMA Line", group="Display")

  // ============================================================
  // FILTER CALCULATIONS
  // ============================================================
  float ema    = ta.ema(close, i_emaLength)
  float avgVol = ta.sma(volume, i_volAvgLength)
  float atr14  = ta.atr(i_atrLength)
  float atrAvg = ta.sma(atr14, i_atrAvgLength)
  bool  inRTH  = not na(time("15", "0930-1600", "America/New_York"))

  bool emaOk     = not i_useEmaFilter     or close > ema
  bool volOk     = not i_useVolFilter     or volume > avgVol
  bool atrOk     = not i_useAtrFilter     or atr14 >= atrAvg * i_atrMinPct
  bool sessionOk = not i_useSessionFilter or inRTH
  bool filtersOk = emaOk and volOk and atrOk and sessionOk

  // ============================================================
  // PIVOT DETECTION (non-repainting)
  // ============================================================
  float swingHigh = ta.pivothigh(high, i_leftBars, i_rightBars)
  float swingLow  = ta.pivotlow(low,  i_leftBars, i_rightBars)

  bool isNewSwingHigh = not na(swingHigh)
  bool isNewSwingLow  = not na(swingLow)

  // ============================================================
  // TRACK CONFIRMED SWING HIGHS AND LOWS
  // ============================================================
  var float prevConfirmedHigh = na
  var float lastConfirmedHigh = na
  var float prevConfirmedLow  = na
  var float lastConfirmedLow  = na

  if isNewSwingHigh
      prevConfirmedHigh := lastConfirmedHigh
      lastConfirmedHigh := swingHigh

  if isNewSwingLow
      prevConfirmedLow := lastConfirmedLow
      lastConfirmedLow := swingLow

  // ============================================================
  // HIGHER LOW / HIGHER HIGH
  // ============================================================
  bool isHigherLow  = isNewSwingLow  and not na(prevConfirmedLow)  and swingLow
  > prevConfirmedLow
  bool isHigherHigh = isNewSwingHigh and not na(prevConfirmedHigh) and swingHigh
  > prevConfirmedHigh

  // ============================================================
  // OPTION A — BREAKOUT STATE MACHINE
  // ============================================================
  var bool  setupActive   = false
  var float breakoutLevel = na
  var bool  breakoutFired = false
  var float lockedHlLevel = na

  if isHigherLow and not na(lastConfirmedHigh)
      setupActive   := true
      breakoutLevel := lastConfirmedHigh
      breakoutFired := false
      lockedHlLevel := swingLow

  bool bullishBreakout = setupActive and not breakoutFired and not
  na(breakoutLevel) and close > breakoutLevel and barstate.isconfirmed

  if bullishBreakout
      breakoutFired := true
      setupActive   := false

  // ============================================================
  // OPTION B — RETEST STATE MACHINE
  // ============================================================
  var bool  retestActive   = false
  var float retestLevel    = na
  var bool  inRetestZone   = false
  var int   retestStartBar = na
  var bool  retestFired    = false

  if bullishBreakout
      retestActive   := true
      retestLevel    := breakoutLevel
      inRetestZone   := false
      retestStartBar := bar_index
      retestFired    := false

  if retestActive
      if (bar_index - retestStartBar) > i_retestTimeout
          retestActive := false
      else if close < retestLevel - atr14
          retestActive := false
      else if not inRetestZone and low <= retestLevel + (atr14 *
  i_retestTolerance)
          inRetestZone := true

  bool retestSignal = retestActive and inRetestZone and not retestFired and close
   > retestLevel and barstate.isconfirmed

  if retestSignal
      retestFired  := true
      retestActive := false

  // ============================================================
  // ENTRY GATE
  // ============================================================
  bool enterBreakout = bullishBreakout and filtersOk and (i_entryMode ==
  "Breakout Only" or i_entryMode == "Both")
  bool enterRetest   = retestSignal    and filtersOk and (i_entryMode == "Retest
  Only"   or i_entryMode == "Both")
  bool enterSignal   = enterBreakout or enterRetest

  // ============================================================
  // RISK MANAGEMENT
  // ============================================================
  var float savedStopPrice = na

  if enterSignal and not na(lockedHlLevel)
      savedStopPrice := lockedHlLevel - (atr14 * i_stopBuffer)

  if enterSignal and strategy.position_size == 0
      strategy.entry("Long", strategy.long)

  if strategy.position_size > 0 and not na(savedStopPrice)
      float entryPrice  = strategy.position_avg_price
      float riskPoints  = entryPrice - savedStopPrice
      float tpPrice     = entryPrice + riskPoints * i_rrRatio
      float trailOffset = atr14 * i_trailAtr
      if i_useTrailStop
          strategy.exit("Exit", "Long", stop=savedStopPrice,
  trail_price=entryPrice, trail_offset=trailOffset / syminfo.mintick)
      else
          strategy.exit("Exit", "Long", stop=savedStopPrice, limit=tpPrice)

  // ============================================================
  // VISUALS — EMA
  // ============================================================
  plot(i_showEma ? ema : na, title="EMA", color=emaOk ? color.new(color.green,
  40) : color.new(color.red, 40), linewidth=1)

  // ============================================================
  // VISUALS — Pivot dots
  // ============================================================
  plotshape(i_showDots and isNewSwingHigh, title="Swing High Dot",
  style=shape.circle, location=location.abovebar, color=color.new(color.red, 30),
   size=size.tiny, offset=-i_rightBars)
  plotshape(i_showDots and isNewSwingLow,  title="Swing Low Dot",
  style=shape.circle, location=location.belowbar, color=color.new(color.teal,
  30), size=size.tiny, offset=-i_rightBars)

  // ============================================================
  // VISUALS — HL / HH labels
  // ============================================================
  if isHigherLow
      label.new(bar_index - i_rightBars, low[i_rightBars] - atr14 * 0.6,
  text="HL", color=color.new(color.lime, 10), textcolor=color.white,
  style=label.style_label_up, size=size.small)

  if isHigherHigh
      label.new(bar_index - i_rightBars, high[i_rightBars] + atr14 * 0.6,
  text="HH", color=color.new(color.red, 10), textcolor=color.white,
  style=label.style_label_down, size=size.small)

  // ============================================================
  // VISUALS — Level lines
  // ============================================================
  plot(setupActive  and not na(breakoutLevel) ? breakoutLevel : na,
  title="Breakout Level", color=color.new(color.orange, 0), linewidth=1,
  style=plot.style_linebr)
  plot(retestActive and not na(retestLevel)   ? retestLevel   : na, title="Retest
   Level",   color=color.new(color.yellow, 0), linewidth=1,
  style=plot.style_linebr)

  // ============================================================
  // VISUALS — Live stop and TP
  // ============================================================
  float liveStop = strategy.position_size > 0 ? savedStopPrice : na
  float liveTp   = strategy.position_size > 0 and not i_useTrailStop ?
  strategy.position_avg_price + (strategy.position_avg_price - savedStopPrice) *
  i_rrRatio : na

  plot(liveStop, title="Stop Loss",   color=color.new(color.red,   20),
  linewidth=1, style=plot.style_linebr)
  plot(liveTp,   title="Take Profit", color=color.new(color.green, 20),
  linewidth=1, style=plot.style_linebr)

  // ============================================================
  // ALERTS
  // ============================================================
  alertcondition(enterBreakout, title="ES - HL-HH Breakout Entry",  message="ES
  Futures 15m - HL-HH Breakout entry. All filters passed.")
  alertcondition(enterRetest,   title="ES - HL-HH Retest Entry",    message="ES
  Futures 15m - HL-HH Retest entry. All filters passed.")
  alertcondition(isHigherLow,   title="ES - Higher Low Confirmed",  message="ES
  Futures 15m - Higher Low confirmed. Setup armed.")
